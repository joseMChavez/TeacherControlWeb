****Vista de Promedio primera Edicion****
create view CalificacionesPromedioView as 
select
 COUNT(Calificaciones.CalificacionId) as Id, Calificaciones.Curso, Calificaciones.Grupo, E.Nombre, E.Matricula, Sum(CD.Puntuacion) as Total, AVG(Calificaciones.TotalPuntos) as Promedio, Calificaciones.Fecha 
  from 
  Calificaciones inner join CalificacionDetalle as CD on Calificaciones.CalificacionId= CD.CalificacionIdM
   inner join Estudiantes as E on E.EstudianteId = Calificaciones.EstudianteIdM  
   inner join Materias as M on M.MateriaId = Calificaciones.MateriaId 
   Group by Calificaciones.CalificacionId, E.Nombre, E.Matricula, Calificaciones.Curso, Calificaciones.Grupo, Calificaciones.Fecha;


   ****Creacion de Vista Estudiantes Por Cursos****
    create view EstudiantePorCursos as
 select
  E.Nombre, E.Matricula,C.Descripcion as Curso, E.Grupo as Seccion, YEar(GETDate())-YEAR(E.FechaNacimiento) as Edad 
  from Estudiantes as E inner join Cursos as C on E.CursoId = C.CursoId;

  ***Creacion de Vista para Estudianes***
  create view View_Estudiantes as
 select E.EstudianteId Id, E.Nombre, E.Matricula,E.Celular,E.Email,E.Direccion,C.Descripcion as Curso, E.Grupo as Seccion, YEar(GETDate())-YEAR(E.FechaNacimiento) as Edad,E.Fecha from Estudiantes as E inner join Cursos as C on E.CursoId = C.CursoId;
